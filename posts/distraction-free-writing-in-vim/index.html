function! s:auto_goyo()
    if &ft == 'markdown'
        Goyo 80
    else
        let bufnr = bufnr('%')
        Goyo!
        execute 'b '.bufnr
    endif
endfunction

augroup goyo_markdown
    autocmd!
    autocmd BufNewFile,BufRead * call s:auto_goyo()
augroup END
