function! ForceHTMLComment(mode, type) range
    set ft=html
    if a:mode == "x"
        execute a:firstline.",".a:lastline."call NERDComment(\"x\", \"".a:type."\")"
    else
        if a:type == "Sexy"
            normal ,cs
        else
            normal ,cc
        endif
    endif
    set ft=php
endfunction
